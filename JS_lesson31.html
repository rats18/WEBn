2. 최종 프로젝트 코드 구조

2.1 완전한 Night & Day 함수

function masterNightDayHandler(self) { // 1. 현재 모드 확인 var currentMode = self.getAttribute('data-mode') || 'day'; // 2. 대상 요소들 선택 var body = document.querySelector('body'); var links = document.querySelectorAll('a'); var buttons = document.querySelectorAll('.mode-button'); // 3. 모드에 따른 스타일 적용 if (currentMode === 'day') { // Night 모드로 전환 body.style.backgroundColor = '#1a1a1a'; body.style.color = '#e0e0e0'; body.style.transition = 'all 0.3s ease'; // 모든 링크 색상 변경 for (var i = 0; i < links.length; i++) { links[i].style.color = '#4FC3F7'; } // 모든 버튼 상태 업데이트 for (var j = 0; j < buttons.length; j++) { buttons[j].setAttribute('data-mode', 'night'); buttons[j].value = 'Day'; buttons[j].style.backgroundColor = '#424242'; buttons[j].style.color = '#ffffff'; } console.log('Night 모드 활성화'); } else { // Day 모드로 전환 body.style.backgroundColor = '#ffffff'; body.style.color = '#333333'; // 모든 링크 색상 변경 for (var i = 0; i < links.length; i++) { links[i].style.color = '#1976D2'; } // 모든 버튼 상태 업데이트 for (var j = 0; j < buttons.length; j++) { buttons[j].setAttribute('data-mode', 'day'); buttons[j].value = 'Night'; buttons[j].style.backgroundColor = '#2196F3'; buttons[j].style.color = '#ffffff'; } console.log('Day 모드 활성화'); } }

2.2 향상된 버전 - 테마 객체 활용

var ThemeManager = { themes: { day: { background: '#ffffff', color: '#333333', linkColor: '#1976D2', buttonBg: '#2196F3', buttonText: 'Night' }, night: { background: '#1a1a1a', color: '#e0e0e0', linkColor: '#4FC3F7', buttonBg: '#424242', buttonText: 'Day' } }, applyTheme: function(themeName) { var theme = this.themes[themeName]; if (!theme) return; // 스타일 적용 document.body.style.backgroundColor = theme.background; document.body.style.color = theme.color; document.body.style.transition = 'all 0.3s ease'; // 링크 색상 변경 var links = document.querySelectorAll('a'); for (var i = 0; i < links.length; i++) { links[i].style.color = theme.linkColor; } // 버튼 업데이트 var buttons = document.querySelectorAll('.mode-button'); for (var j = 0; j < buttons.length; j++) { buttons[j].value = theme.buttonText; buttons[j].style.backgroundColor = theme.buttonBg; buttons[j].setAttribute('data-current-theme', themeName); } }, toggle: function(button) { var currentTheme = button.getAttribute('data-current-theme') || 'day'; var nextTheme = currentTheme === 'day' ? 'night' : 'day'; this.applyTheme(nextTheme); } };
